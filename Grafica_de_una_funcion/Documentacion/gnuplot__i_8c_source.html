739 char const * gnuplot_tmpfile(gnuplot_ctrl * handle)
740 {
741  static char const * tmp_filename_template = "gnuplot_tmpdatafile_XXXXXX";
742  char * tmp_filename = NULL;
743  int tmp_filelen = strlen(tmp_filename_template);
744 
745 #ifndef WIN32
746  int unx_fd;
747 #endif // #ifndef WIN32
748 
749  assert(handle->tmp_filename_tbl[handle->ntmp] == NULL);
750 
751  /* Open one more temporary file? */
752  if (handle->ntmp == GP_MAX_TMP_FILES - 1) {
753  fprintf(stderr,
754  "maximum # of temporary files reached (%d): cannot open more",
755  GP_MAX_TMP_FILES) ;
756  return NULL;
757  }
758 
759  tmp_filename = (char*) malloc(tmp_filelen+1);
760  if (tmp_filename == NULL)
761  {
762  return NULL;
763  }
764  strcpy(tmp_filename, tmp_filename_template);
765 
766 #ifdef WIN32
767  if (_mktemp(tmp_filename) == NULL)
768  {
769  return NULL;
770  }
771 #else // #ifdef WIN32
772  unx_fd = mkstemp(tmp_filename);
773  if (unx_fd == -1)
774  {
775  return NULL;
776  }
777  close(unx_fd);
778 
779 #endif // #ifdef WIN32
780 
781  handle->tmp_filename_tbl[handle->ntmp] = tmp_filename;
782  handle->ntmp ++;
783  return tmp_filename;
784 }
